    let svgCircles = document.querySelectorAll("circle");

setInterval(() => {
  svgCircles.forEach(circle => {
    if (circle.getAttribute('fill') === 'white') {
      circle.setAttribute('fill', 'black');
    } else {
      circle.setAttribute('fill', 'white');
    }
  });
}, 1000);
